    # 获取PID

    // format: "pid@hostname"
String HOST_NAME = ManagementFactory.getRuntimeMXBean().getName(); 
public static int getPid() {
    try {
        return Integer.parseInt(HOST_NAME.substring(0, HOST_NAME.indexOf('@')));
    } catch (Exception e) {
        return -1;
    }
}
